#include<iostream>
class sum
{
int z;
public:void input(int,int);
	void output();
};

void sum::input(int a,int b)
{ 
z=b-a;
}

void sum::output()
{
std::cout<<z<<std::endl;
}

int main()
{
int t,x,y;
sum s[20];
std::cin>>t;
for(int i=0;i<t;i++)
{
        std::cin>>x>>y;
        s[i].input(x,y);
        }
        
        for(int j=0;j<t;j++)
        s[j].output();   
system("pause");
return 0;
}
