#include<iostream>
using namespace std;
class f
{
int z;
public:f();
       void input();
	void output();
};
f::f()
{z=2;
          }

void f::input()
{ 
z=z*z;
}

void f::output()
{
cout<<z<<endl;
}

int main()
{
int n,x,y,i,j;
f fo[20];
cin>>n;
for(i=0;i<n;i++)
{
        cin>>x>>y;
        for(j=0;j<x;j++)
        fo[i].input();
        }
        for(i=0;i<n;i++)
        fo[i].output();
system("pause");
return 0;
}
