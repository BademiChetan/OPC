#include<iostream>
class pro
{
      int N,a[100];
      public:void input();
             void output();
      };
void pro::input()
{
     int i;
     std::cin>>N;
     std::cout<<"\n";
     for(i=0;i<N;i++)
     std::cin>>a[i];
     }
void pro::output()
{
     int sum=0;
     for(int i=0;i<N;i++)
     {
             sum=sum+a[i];
             }
     sum=sum-1;
     if(sum%2==0)
     std::cout<<"LOSE\n";
     else
     std::cout<<"WIN\n";
}    
     
int main(void)
{
    pro k[100];
    int t,i;
    std::cin>>t;
    for(i=0;i<t;i++)
    k[i].input();
    std::cout<<"\n";
    for(i=0;i<t;i++)
    k[i].output();
    std::cout<<"\n";
    system("pause");
    return 0;
}
